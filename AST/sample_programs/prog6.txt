class Program{
    int abc[10];

    void print_val(int a)
    {
        callout("printf", "%d ", a);
        callout("puts", "");
        return;
    }
    
    int main(){
        int a, i;

        for i=0, 10
        {
            abc[i] = i;
        }
        for i=0, 10
        {
            callout("printf", "%d ", abc[i]);
            print_val(abc[i]);
        }
        callout("puts", "");
        return 0;
    }

}
