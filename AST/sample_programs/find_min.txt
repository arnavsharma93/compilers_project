	int a[10];
	
	int main()
	{
		int i, min_value;
		for i=0, 10
		{
			a[i] = (callout("rand") %20);
			if(i==0)
			{
				min_value = a[i];
			}
			else
			{
				if(min_value > a[i])
				{
					min_value = a[i];
				}
			}
		}
	
		callout("puts", "array A");
		for i=0, 10
		{
			callout("printf", "%d ", a[i]);
		}
		callout("puts", "");

		callout("printf", "Min value %d", min_value);
		callout("puts", "");

		return 0;
	}
}
